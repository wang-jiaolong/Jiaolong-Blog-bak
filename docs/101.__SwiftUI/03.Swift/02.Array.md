## 遍历

```swift
// 下标遍历
for i in 0..<array.count{
	print(array[i])
}

// 直接遍历
for s in array{
	print(s)
}

//反转
for s in array.reversed(){ 
	print(s)
}
```

## Extension

```swift
extension Array where Element: Equatable {
 
mutating func remove(_ object: Element) {
    if let index = index(of: object) {
        remove(at: index)
    }
}}
```
