# Basic

```swift
struct ContentView: View {
    var body: some View {
        TabView {
            Text("Bananas 🍌🍌")
                .tabItem {
                    Image(systemName: "1.circle.fill")
                    Text("🍌🍌")
                }
            Text("Apples 🍏🍏")
                .tabItem {
                    Image(systemName: "2.square.fill")
                    Text("🍏🍏")
                }
            Text("Peaches 🍑🍑")
                .tabItem {
                    Image(systemName: "3.square.fill")
                    Text("🍑🍑")
                }
        }
        .font(.headline)
    }
}
```

# `.tabViewStyle`

```swift
        .tabViewStyle(PageTabViewStyle())
```

```swift
        .tabViewStyle(PageTabViewStyle(indexDisplayMode: .never))
```
